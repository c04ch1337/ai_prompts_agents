### PART 4 – Desktop Agent Code (Pure Rust – Connects to Everything)

```rust
// desktop/src/client/unified_client.rs
use reqwest::Client;
use serde_json::json;
use tokio_tungstenite::{connect_async, tungstenite::protocol::Message};
use futures_util::{SinkExt, StreamExt};

pub struct UnifiedDesktopClient {
    http: Client,
    ws: Option<tokio_tungstenite::WebSocketStream<tokio_tungstenite::MaybeTlsStream<tokio::net::TcpStream>>>,
}

impl UnifiedDesktopClient {
    pub async fn new() -> Self {
        let cert = std::fs::read("certs/desktop_cert.pem").unwrap();
        let key = std::fs::read("certs/desktop_key.pem").unwrap();
        let ca = std::fs::read("certs/gateway_ca.pem").unwrap();

        let client = Client::builder()
            .identity(reqwest::Identity::from_pem(&[cert, key].concat()).unwrap())
            .add_root_certificate(reqwest::Certificate::from_pem(&ca).unwrap())
            .danger_accept_invalid_certs(false)
            .build()
            .unwrap();

        // Connect to Warp A2A bus
        let (ws, _) = connect_async("wss://127.0.0.1:4000/warp/ws")
            .await
            .unwrap();

        Self { http: client, ws: Some(ws) }
    }

    pub async fn ask(&self, task: &str) -> String {
        let resp = self.http.post("https://127.0.0.1:4000/api/v1/orchestrate")
            .json(&json!({ "task": task }))
            .send()
            .await
            .unwrap()
            .json::<serde_json::Value>()
            .await
            .unwrap();
        resp["result"].as_str().unwrap_or("Done").to_string()
    }

    pub async fn spawn_swarm(&self, kind: &str, goal: &str) -> String {
        self.ask(&format!("Spawn a {kind} swarm to {goal}")).await
    }

    pub async fn listen_a2a(&mut self, callback: impl Fn(String)) {
        while let Some(msg) = self.ws.as_mut().unwrap().next().await {
            if let Ok(Message::Text(text)) = msg {
                callback(text);
            }
        }
    }
}
```
